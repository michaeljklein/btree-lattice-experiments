Equations canonical_btree (n : nat) : @btree 1 :=
canonical_btree O := bnil tt;
canonical_btree (S n) := bcons (bnil tt) (canonical_btree n).

Fixpoint num_leaves_canonical_btree (n : nat) :
  num_leaves (canonical_btree n) = S n.
