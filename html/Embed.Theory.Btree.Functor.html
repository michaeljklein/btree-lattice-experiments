Definition void {f : @Functor Coq Coq} {x} : f x ~> f 1 := fmap one.

Equations fmap_btree {A B : Type} (f : A → B) : @btree A → @btree B :=
fmap_btree f (bnil x) := bnil (f x) ;
fmap_btree f (bcons x y) := bcons (fmap_btree f x) (fmap_btree f y).
