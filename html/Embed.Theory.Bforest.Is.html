Definition is_bforest {A : Type} (n m : nat) (xs : list (@btree A)) : Prop :=
  n = length xs ∧ m = list_num_leaves xs.

Lemma eq_is_bforest_btree {A} {m} (xs ys : list (@btree A))

  (pf : xs = ys) :
    is_bforest_btree xs xss = is_bforest_btree ys yss.
